## Usage

```tsx showLineNumbers

```
 
```tsx showLineNumbers
export default function Page() {
  return (
    <div>
      <GradientMesh
        colors={["#3b2a8d", "#aaa7d7", "#f75092"]}
        distortion={5}
        swirl={0.5}
        speed={1}
        scale={1}
        offsetX={0}
        offsetY={0}
        rotation={90}
        waveAmp={0.1}
        waveFreq={10}
        waveSpeed={0.2}
        grain={0.06}
      />
    </div>
  );
}
``` 
